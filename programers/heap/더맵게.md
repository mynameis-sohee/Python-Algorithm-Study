def solution(scoville, K):
    scoville.sort()
    import heapq as hq
    answer = 0
    
    while True:
        a = scoville[0] + 2 * scoville[1]
        hq.heappushpop(scoville, a)
        hq.heappop(scoville)
        answer += 1
        if min(scoville) >= K:
            break
    else:
        answer = -1
    
    return answer
