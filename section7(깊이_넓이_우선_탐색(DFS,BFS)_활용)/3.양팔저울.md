## 모범 답안

```python
import sys
sys.stdin = open("input.txt", "r")
def DFS(L, sum):
    global res
    if L == n:
        if 0 < sum <=s: #음수인 경우는 보지 않는다. 왜냐하면 양수로 쌍이 있기 때문
            res.add(sum)

    else:
        DFS(L+1, sum+G[L]) #왼쪽에 둔다
        DFS(L+1, sum-G[L]) #추를 오른쪽에 둔다
        DFS(L+1, sum) #추를 쓰지 않는다

if __name__ == '__main__':
    n = int(input())
    G = list(map(int, input().split()))
    s = sum(G)
    res = set()
    DFS(0,0)
    print(s-len(res))
```
