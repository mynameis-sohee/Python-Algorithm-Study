import sys
sys.stdin = open("in1.txt",'r')
n = int(input())
arr = list(map(int, input().split()))
arr.insert(0,0)
dy =[0] *(n+1)
dy[1] = 1 #배열에 원소 하나일 때
res = 0

for i in range(2, n+1):
    maxnum = 0
    for j in range(i-1, 0,-1): #배열 앞으로 탐색 #마지막 숫자의 앞의 값 탐색
        if arr[j] < arr[i] and dj[j] > maxnum: #i번째가 배열의 마지막 숫자.j를 찾는 것
            maxnum = dy[j]
    dy[i] = maxnum + 1
    if dy[i] > res: #dy 테이블의 가장 큰 값이 답
        res = dy[i]
